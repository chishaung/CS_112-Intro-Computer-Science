CODE

breed [humans human]
breed [zombies zombie]




to setup
  set-default-shape zombies "zombies"
  set-default-shape humans "person"  
  clear-all
  create-humans Numbers_of_Humans [
    set size 1.5
    setxy random-xcor random-ycor
    set color white
  ]
  create-zombies Numbers_of_Zombies [
    set size 3
    setxy random-xcor random-ycor
    set color green
  ]
end

to go
  move-humans
  move-zombies
  do-plots
  ;; combat
end

to move-humans
  ask humans [set heading towards min-one-of zombies[distance myself]]
  ask humans [set heading (heading + 180)]
  ask humans [fd 0.4]
end

to move-zombies
  if count humans > 0
     [ask zombies [set heading towardsxy random-xcor random-ycor]]
  ask zombies [fd 0.2]
  if count humans > 0
    [ask zombies [ask humans-here [set breed zombies set color green]]]
end

to do-plots 
 set-current-plot "Populations" 
 set-current-plot-pen "Humans" 
 plot count humans 
 set-current-plot-pen "Zombies" 
 plot count zombies
end
